    fn run_repl_iteration(home: &'_ mut SmartHome)
        ->
            Result<SessionOutcome> {
        { ::std::io::_print(format_arguments::from_str("smarthome> ")); };
        match branch(io::stdout().flush()) {
            Break {  0: residual } => #[allow(unreachable_code)]
                return from_residual(residual),
            Continue {  0: val } => #[allow(unreachable_code)]
                val,
        };
        let mut line = String::new();
        let bytes =
            match branch(io::stdin().read_line(&mut line)) {
                Break {  0: residual } => #[allow(unreachable_code)]
                    return from_residual(residual),
                Continue {  0: val } => #[allow(unreachable_code)]
                    val,
            };
        { ::std::io::_print(format_arguments::from_str("\n")); };
        if bytes == 0 { return Ok(SessionOutcome::Exit); }
        let trimmed = line.trim();
        if trimmed.is_empty() { return Ok(SessionOutcome::Continue); }
        if trimmed.eq_ignore_ascii_case("exit") ||
                trimmed.eq_ignore_ascii_case("quit") {
            return Ok(SessionOutcome::Exit);
        }
        if trimmed.eq_ignore_ascii_case("help") {
            print_help();
            return Ok(SessionOutcome::Continue);
        }
        if trimmed.eq_ignore_ascii_case("status") {
            {
                ::std::io::_print({
                        super let args = (&home.render_status(),);
                        super let args = [format_argument::new_display(args.0)];
                        unsafe { format_arguments::new(b"\xc0\x01\n\x00", &args) }
                    });
            };
            { ::std::io::_print(format_arguments::from_str("\n")); };
            return Ok(SessionOutcome::Continue);
        }
        let cmd =
            match branch(parse_command(trimmed)) {
                Break {  0: residual } => #[allow(unreachable_code)]
                    return from_residual(residual),
                Continue {  0: val } => #[allow(unreachable_code)]
                    val,
            };
        home.apply(cmd);
        {
            ::std::io::_print({
                    super let args = (&home.render_status(),);
                    super let args = [format_argument::new_display(args.0)];
                    unsafe { format_arguments::new(b"\xc0\x01\n\x00", &args) }
                });
        };
        { ::std::io::_print(format_arguments::from_str("\n")); };
        Ok(SessionOutcome::Continue)
    }
    fn parse_command(input: &'_ str)
        ->
            Result<Command> {
        let parts: Vec<&str> = input.split_whitespace().collect();
        if parts.is_empty() {
            return Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Leerer Command"));
                            error
                        }));
        }
        match parts[0].to_lowercase().as_str() {
            "heating" => parse_heating(&parts),
            "lights" => parse_lights(&parts),
            "security" => parse_security(&parts),
            _ =>
                Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Unbekannter Command. Nutze \'help\' f\u{fc}r alle verf\u{fc}gbaren Befehle."));
                            error
                        })),
        }
    }
    fn parse_heating(parts: &'_ [&'_ str])
        ->
            Result<Command> {
        if parts.len() < 3 {
            return Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: heating set kitchen 22"));
                            error
                        }));
        }
        match parts[1].to_lowercase().as_str() {
            "set" => {
                if parts.len() != 4 {
                    return Err(::anyhow::__private::must_use({
                                    let error =
                                        ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: heating set kitchen 22"));
                                    error
                                }));
                }
                let where_ =
                    match branch(RoomSelection::parse(parts[2])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                let temp_value: f32 =
                    match branch(parts[3].parse().map_err(|_|
                                    ::anyhow::__private::must_use({
                                            let error =
                                                ::anyhow::__private::format_err(format_arguments::from_str("Temperatur ist keine g\u{fc}ltige Zahl"));
                                            error
                                        }))) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                let target =
                    match branch(TemperatureC::new(temp_value).ok_or_else(||
                                    ::anyhow::__private::must_use({
                                            let error =
                                                ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Temperatur. Erlaubt: 5.0\u{b0}C bis 35.0\u{b0}C"));
                                            error
                                        }))) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                Ok(Command::HeatingSet { where_, target })
            }
            "on" => {
                if parts.len() != 3 {
                    return Err(::anyhow::__private::must_use({
                                    let error =
                                        ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: heating on all"));
                                    error
                                }));
                }
                let where_ =
                    match branch(RoomSelection::parse(parts[2])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                Ok(Command::HeatingEnabled { where_, enabled: true })
            }
            "off" => {
                if parts.len() != 3 {
                    return Err(::anyhow::__private::must_use({
                                    let error =
                                        ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: heating off bedroom"));
                                    error
                                }));
                }
                let where_ =
                    match branch(RoomSelection::parse(parts[2])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                Ok(Command::HeatingEnabled { where_, enabled: false })
            }
            _ =>
                Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Unbekannter heating-Befehl. Nutze: set/on/off"));
                            error
                        })),
        }
    }
    fn parse_lights(parts: &'_ [&'_ str])
        ->
            Result<Command> {
        if parts.len() < 3 {
            return Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiele: lights set living on | lights toggle all"));
                            error
                        }));
        }
        match parts[1].to_lowercase().as_str() {
            "set" => {
                if parts.len() != 4 {
                    return Err(::anyhow::__private::must_use({
                                    let error =
                                        ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: lights set living on"));
                                    error
                                }));
                }
                let where_ =
                    match branch(RoomSelection::parse(parts[2])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                let state =
                    match branch(OnOff::parse(parts[3])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]

            "toggle" => {
                if parts.len() != 3 {
                    return Err(::anyhow::__private::must_use({
                                    let error =
                                        ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: lights toggle bath"));
                                    error
                                }));
                }
                let where_ =
                    match branch(RoomSelection::parse(parts[2])) {
                        Break {  0: residual } => #[allow(unreachable_code)]
                            return from_residual(residual),
                        Continue {  0: val } => #[allow(unreachable_code)]
                            val,
                    };
                Ok(Command::LightsToggle { where_ })
            }
            _ =>
                Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Unbekannter lights-Befehl. Nutze: set/toggle"));
                            error
                        })),
        }
    }
    fn parse_security(parts: &'_ [&'_ str])
        ->
            Result<Command> {
        if parts.len() != 2 {
            return Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Ung\u{fc}ltige Eingabe. Beispiel: security lock | security unlock"));
                            error
                        }));
        }
        match parts[1].to_lowercase().as_str() {
            "lock" => Ok(Command::SecurityLockAll),
            "unlock" => Ok(Command::SecurityUnlockAll),
            _ =>
                Err(::anyhow::__private::must_use({
                            let error =
                                ::anyhow::__private::format_err(format_arguments::from_str("Unbekannter security-Befehl. Nutze: lock/unlock"));
                            error
                        })),
        }
    }

mod storage {
    use crate::domain::SmartHome;
    use ::{};
    use anyhow::Context;
    use anyhow::Result;
    use std::fs;
    use std::path::Path;
    const STATE_FILE: &'_ str = "smarthome_state.json";
    fn load_or_default()
        ->
            Result<SmartHome> {
        if !Path::new(STATE_FILE).exists() {
            return Ok(SmartHome::new_default());
        }
        let json =
            match branch(fs::read_to_string(STATE_FILE).context("Konnte State-Datei nicht lesen"))
                {
                Break {  0: residual } => #[allow(unreachable_code)]

            match branch(serde_json::from_str(&json).context("State-Datei ist ung\u{fc}ltiges JSON"))
                {

                    return from_residual(residual),
                Continue {  0: val } => #[allow(unreachable_code)]
                    val,
            };
        Ok(home)
    }
    fn save(home: &'_ SmartHome)

        let json =
            match branch(serde_json::to_string_pretty(home).context("Konnte State nicht serialisieren"))

    }
